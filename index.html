4. Trigger Database

create database man;
use man;

create table customers (
id int primary key,
name varchar(100),
age int,
address varchar(100),
salary decimal(10,2)
);

create table salary_audit (
logid int auto_increment primary key,
custid int,
actiontype varchar(50),
old_salary decimal(10,2),
new_salary decimal(10,2),
salary_diff decimal(10,2),
changedate datetime
);

desc salary_audit;

delimiter $$

create trigger custid_insert_audit
after insert on customers
for each row
begin
insert into salary_audit (custid, actiontype, old_salary, new_salary, salary_diff, changedate)
values (new.id, 'insert', null, new.salary, null, now());
end$$

create trigger custid_delete_audit
after delete on customers
for each row
begin
insert into salary_audit (custid, actiontype, old_salary, new_salary, salary_diff, changedate)
values (old.id, 'delete', old.salary, null, null, now());
end$$

create trigger custid_update_audit
after update on customers
for each row
begin
insert into salary_audit (custid, actiontype, old_salary, new_salary, salary_diff, changedate)
values (old.id, 'update', old.salary, new.salary, new.salary - old.salary, now());
end$$

delimiter ;
