Minimum Spanning Tree — Kruskal's Algorithm
Greedy · Union-Find · O(E log E)
▾ 
#include<stdio.h>

int find(int parent[], int i)
{
    if(parent[i] == i)
        return i;

    return find(parent, parent[i]);
}

int main()
{
    int n, i;

    printf("Enter number of edges: ");
    scanf("%d", &n);

    int edges[n][3];

    printf("Enter edges and weights:\n");

    for(i = 0; i < n; i++)
    {
        scanf("%d%d%d", &edges[i][0], &edges[i][1], &edges[i][2]);
    }

    int parent[4] = {0,1,2,3};

    int cost = 0, count = 0;

    printf("Edges in Minimum Spanning Tree:\n");

    for(i = 0; i < n && count < 3; i++)
    {
        int u = edges[i][0];
        int v = edges[i][1];

        int pu = find(parent, u);
        int pv = find(parent, v);

        if(pu != pv)
        {
            parent[pu] = pv;

            printf("(%d,%d) : %d\n", u, v, edges[i][2]);

            cost = cost + edges[i][2];
            count++;
        }
    }

    printf("Minimum cost = %d\n", cost);

    return 0;
}

Minimum Spanning Tree — Prim's Algorithm
Greedy · Cost Matrix · O(V²)
▾ 
#include<stdio.h>

int main()
{
    int cost[10][10], visited[10]={0};
    int i, j, n, edges=0, min, x, y, total=0;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter cost matrix:\n");

    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            scanf("%d", &cost[i][j]);
        }
    }

    visited[0] = 1;

    printf("Edges in MST:\n");

    while(edges < n-1)
    {
        min = 999;

        for(i=0; i<n; i++)
        {
            if(visited[i])
            {
                for(j=0; j<n; j++)
                {
                    if(!visited[j] && cost[i][j]!=0)
                    {
                        if(cost[i][j] < min)
                        {
                            min = cost[i][j];
                            x = i;
                            y = j;
                        }
                    }
                }
            }
        }

        printf("%d-%d = %d\n", x, y, min);

        total = total + min;
        visited[y] = 1;
        edges++;
    }

    printf("Total weight = %d", total);

    return 0;
}

All-Pairs Shortest Path — Floyd-Warshall
Dynamic Programming · O(V³)
▾ 
#include<stdio.h>

int main()
{
    int a[10][10], n, i, j, k;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter cost of matrix:\n");

    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            scanf("%d", &a[i][j]);
        }
    }

    for(k=0; k<n; k++)
    {
        for(i=0; i<n; i++)
        {
            for(j=0; j<n; j++)
            {
                if(a[i][j] > a[i][k] + a[k][j])
                {
                    a[i][j] = a[i][k] + a[k][j];
                }
            }
        }
    }

    printf("\nThe final distance matrix is:\n");

    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            printf("%d\t", a[i][j]);
        }

        printf("\n");
    }

    return 0;
}

Transitive Closure — Warshall's Algorithm
Dynamic Programming · O(V³)
▾ 
#include<stdio.h>

int main()
{
    int a[10][10], n, i, j, k;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");

    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            scanf("%d", &a[i][j]);
        }
    }

    for(k=0; k<n; k++)
    {
        for(i=0; i<n; i++)
        {
            for(j=0; j<n; j++)
            {
                a[i][j] = a[i][j] || (a[i][k] && a[k][j]);
            }
        }
    }

    printf("\nThe final transitive matrix is:\n");

    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            printf("%d\t", a[i][j]);
        }

        printf("\n");
    }

    return 0;
}

Topological Sorting
Graph · In-degree / Kahn's · O(V + E)
▾ 
#include<stdio.h>

int main()
{
    int a[10][10], in[10]={0}, s[10], t[10];
    int n, i, j, top=-1, k=0;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");

    for(i=0; i<n; i++)
        for(j=0; j<n; j++)
        {
            scanf("%d", &a[i][j]);
            in[j] += a[i][j];
        }

    for(i=0; i<n; i++)
        if(in[i] == 0)
            s[++top] = i;

    while(top != -1)
    {
        i = s[top--];
        t[k++] = i + 1;

        for(j=0; j<n; j++)
            if(a[i][j])
            {
                in[j]--;

                if(in[j] == 0)
                    s[++top] = j;
            }
    }

    printf("Topological Order : ");

    for(i=0; i<k; i++)
        printf("%d ", t[i]);

    return 0;
}

Selection Sort with Execution Time
Sorting · Comparison-based · O(N²)
▾ 
#include <stdio.h>
#include <time.h>

int main()
{
    int a[10000], n, i, j, min, temp;
    clock_t start, end;
    float time;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter elements:\n");

    for(i = 0; i < n; i++)
    {
        scanf("%d", &a[i]);
    }

    start = clock();

    for(i = 0; i < n - 1; i++)
    {
        min = i;

        for(j = i + 1; j < n; j++)
        {
            if(a[j] < a[min])
            {
                min = j;
            }
        }

        temp = a[i];
        a[i] = a[min];
        a[min] = temp;
    }

    end = clock();

    time = (float)(end - start) / CLOCKS_PER_SEC;

    printf("Sorted elements are:\n");

    for(i = 0; i < n; i++)
    {
        printf("%d ", a[i]);
    }

    printf("\nTime taken = %f seconds", time);

    return 0;
}
